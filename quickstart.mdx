# Quick start

This guide will help you create your first Data Command with nested items and async loading.

<Steps>
  <Step title="Create a basic command palette">
    Start with a simple command palette wrapped in a Dialog:

    ```tsx
    "use client";

    import { Button } from "@/components/ui/button";
    import { Dialog, DialogContent, DialogTitle } from "@/components/ui/dialog";
    import { DialogTrigger } from "@radix-ui/react-dialog";
    import { DataCommand } from "@/components/data-command";

    export default function App() {
      return (
        <Dialog>
          <DialogTrigger asChild>
            <Button>Search Site</Button>
          </DialogTrigger>
          <DialogContent className="p-0">
            <DialogTitle className="sr-only">Site Search</DialogTitle>
            <DataCommand items={[]} />
          </DialogContent>
        </Dialog>
      );
    }
    ```
  </Step>

  <Step title="Add items with icons">
    Add some static items to your command palette:

    ```tsx
    import { User, BanknoteIcon } from "lucide-react";

    <DataCommand
      items={[
        {
          icon: <User />,
          label: "Profile",
          value: "profile",
          onSelect: () => console.log("Profile selected"),
        },
        {
          icon: <BanknoteIcon />,
          label: "Billing",
          value: "billing",
          onSelect: () => console.log("Billing selected"),
        },
      ]}
    />
    ```

    Each item needs:
    - `label` - What the user sees
    - `value` - Unique identifier for the item
    - `onSelect` - Callback when the item is selected (optional)
    - `icon` - Icon to display (optional)
  </Step>

  <Step title="Add nested items with loadItems">
    Make items expandable by adding a `loadItems` function:

    ```tsx
    import { SettingsIcon } from "lucide-react";

    <DataCommand
      items={[
        {
          icon: <User />,
          label: "Profile",
          value: "profile",
          loadItems: async () => [
            {
              label: "Settings",
              value: "settings",
              icon: <SettingsIcon />,
              onSelect: () => alert("You have selected: profile/settings"),
            },
          ],
        },
      ]}
    />
    ```

    Users can press **Tab** to drill down into nested items, or **Backspace** to go back up.
  </Step>

  <Step title="Add close behavior">
    Use the `onClose` prop to close the dialog when an item is selected:

    ```tsx
    export default function App() {
      const [open, setOpen] = useState(false);

      return (
        <Dialog open={open} onOpenChange={setOpen}>
          <DialogTrigger asChild>
            <Button>Search Site</Button>
          </DialogTrigger>
          <DialogContent className="p-0">
            <DialogTitle className="sr-only">Site Search</DialogTitle>
            <DataCommand
              items={[/* ... */]}
              onClose={() => setOpen(false)}
            />
          </DialogContent>
        </Dialog>
      );
    }
    ```

    The command palette will automatically call `onClose` when a user selects an item with `onSelect`.
  </Step>
</Steps>

## Complete example

Here's everything together:

```tsx
"use client";

import { useState } from "react";
import { Button } from "@/components/ui/button";
import { Dialog, DialogContent, DialogTitle } from "@/components/ui/dialog";
import { DialogTrigger } from "@radix-ui/react-dialog";
import { BanknoteIcon, SettingsIcon, User } from "lucide-react";
import { DataCommand } from "@/components/data-command";

export default function App() {
  const [open, setOpen] = useState(false);

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>
        <Button>Search Site</Button>
      </DialogTrigger>
      <DialogContent className="p-0">
        <DialogTitle className="sr-only">Site Search</DialogTitle>
        <DataCommand
          items={[
            {
              icon: <User />,
              label: "Profile",
              value: "profile",
              loadItems: async () => [
                {
                  label: "Settings",
                  value: "settings",
                  icon: <SettingsIcon />,
                  onSelect: () => alert("You have selected: profile/settings"),
                },
              ],
            },
            {
              icon: <BanknoteIcon />,
              label: "Billing",
              value: "billing",
              onSelect: () => alert("You have selected: billing"),
            },
          ]}
          onClose={() => setOpen(false)}
        />
      </DialogContent>
    </Dialog>
  );
}
```

<Tip>
  Press **Tab** to drill down into nested items, **Backspace** to go back, and **Enter** to select the current item.
</Tip>
